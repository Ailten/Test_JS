DROP TABLE IF EXISTS `user`;
CREATE TABLE `user` (
	id int NOT NULL AUTO_INCREMENT,
	name varchar(32) UNIQUE,
	PRIMARY KEY (id)
);

DROP TABLE IF EXISTS `vod`;
CREATE TABLE `vod` (
	id int NOT NULL AUTO_INCREMENT,
    	idUser int NOT NULL,
   	name varchar(32),
    	PRIMARY KEY (id),
    	FOREIGN KEY (idUser) REFERENCES `user`(id)
);
